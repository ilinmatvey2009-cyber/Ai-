"""
AI-парсер заявки: превращает произвольный текст ("труба нержа 20 полтора 3 метра")
в структурированные атрибуты товара.

В ПРОДАКШЕНЕ: реальный вызов LLM (пример — Claude API).
Стоимость: Claude Haiku ~ $0.001-0.003 за заявку, задержка ~0.5-1 сек.
"""
import os, json, re

DEMO_MODE = True  # False -> реальный вызов API (нужен ANTHROPIC_API_KEY + доступ в сеть)

EXTRACTION_PROMPT = """Ты — парсер заявок для отдела снабжения.
Извлеки из текста заявки клиента структурированные данные в JSON:
{
  "product_type": "базовый тип товара (труба/кабель/профиль/крепёж/инструмент/...)",
  "material": "материал, если указан (нержавеющая сталь/оцинкованный/ПВХ/алюминий/...) или null",
  "dimensions": "размеры в формате AxB или одно число, если указаны, иначе null",
  "quantity": число или null,
  "unit": "единица измерения (м/шт/кг/т/м2/меш) или null",
  "raw_normalized": "нормализованное короткое название товара для поиска"
}
Текст заявки: "{text}"
Верни только JSON, без пояснений."""

def ai_extract(raw_text: str) -> dict:
    if DEMO_MODE:
        return _mock_llm_extract(raw_text)
    else:
        import anthropic
        client = anthropic.Anthropic(api_key=os.environ["ANTHROPIC_API_KEY"])
        resp = client.messages.create(
            model="claude-haiku-4-5",
            max_tokens=300,
            messages=[{"role": "user", "content": EXTRACTION_PROMPT.format(text=raw_text)}]
        )
        return json.loads(resp.content[0].text)

def _mock_llm_extract(raw_text: str) -> dict:
    """Заглушка для офлайн-демонстрации логики (в проде заменяется реальным вызовом LLM)."""
    t = raw_text.lower()
    material = None
    for kw, norm in [("нерж", "нержавеющая сталь"), ("оцинк", "оцинкованный"),
                     ("пвх", "ПВХ"), ("алюмин", "алюминий"), ("медн", "медь")]:
        if kw in t:
            material = norm
            break
    product_type = None
    for kw in ["труба", "кабель", "провод", "профиль", "лист", "арматура",
               "уголок", "швеллер", "саморез", "болт", "дюбель", "гайка",
               "перфоратор", "болгарк", "шурупов", "шурik", "шурик", "уровень"]:
        if kw in t:
            product_type = kw
            break
    dims = re.findall(r"\d+[.,]?\d*\s*(?:х|x|на)\s*\d+[.,]?\d*", t)
    dimensions = dims[0].replace(" ", "").replace(",", ".") if dims else None
    if not dimensions and "полтора" in t:
        nums = re.findall(r"\d+", t)
        if nums:
            dimensions = f"{nums[0]}x1.5"
    qty_match = re.search(r"(\d+)\s*(шт|метр|м\b|кг|т\b)", t)
    quantity = int(qty_match.group(1)) if qty_match else None
    unit = qty_match.group(2) if qty_match else None
    unit = {"метр": "м"}.get(unit, unit)

    return {
        "product_type": product_type, "material": material, "dimensions": dimensions,
        "quantity": quantity, "unit": unit, "raw_normalized": raw_text
    }

"""
Матчер заявок к базе прайсов поставщиков.
Масштабирование на 200 000 SKU: обратный индекс по токенам строится один раз при
загрузке базы; в проде — PostgreSQL + pg_trgm или Elasticsearch/OpenSearch/Meilisearch.
"""
import re
from difflib import SequenceMatcher

SYNONYMS = {"нерж": "нержавеющая", "нержавейка": "нержавеющая",
            "проф": "профильная", "оцинковка": "оцинкованный"}

def normalize(text: str):
    t = text.lower().replace("х", "x")
    for k, v in SYNONYMS.items():
        t = t.replace(k, v)
    t = re.sub(r"[^\w\.]+", " ", t)
    return [tok for tok in t.split() if tok]

def build_index(products: list):
    index, norm_cache = {}, []
    for i, name in enumerate(products):
        toks = normalize(name)
        norm_cache.append(toks)
        for t in toks:
            index.setdefault(t, set()).add(i)
    return index, norm_cache

def match_query(query, products, index, norm_cache, ai_attrs=None, catalog_attrs=None, top_n=8):
    q_tokens = normalize(query)
    scores = {}
    for t in q_tokens:
        for pid in index.get(t, []):
            scores[pid] = scores.get(pid, 0) + 1
    candidates = sorted(scores, key=scores.get, reverse=True)[:top_n] if scores else range(len(products))
    best_pid, best_score = None, -1
    q_str = " ".join(q_tokens)
    for pid in candidates:
        p_str = " ".join(norm_cache[pid])
        ratio = SequenceMatcher(None, q_str, p_str).ratio()
        overlap = scores.get(pid, 0)
        bonus = 0
        if ai_attrs and catalog_attrs:
            if ai_attrs.get("dimensions") and catalog_attrs[pid].get("dimensions") == ai_attrs["dimensions"]:
                bonus += 0.5
            if ai_attrs.get("material") and catalog_attrs[pid].get("material") == ai_attrs["material"]:
                bonus += 0.3
        final = ratio * 0.5 + (overlap / max(len(q_tokens), 1)) * 0.1 + bonus
        if final > best_score:
            best_score, best_pid = final, pid
    return best_pid, round(best_score, 3)

def top3_suppliers(rows_for_product):
    rows = sorted(rows_for_product, key=lambda r: r[1])
    prices = [r[1] for r in rows]
    min_price, avg_price = min(prices), sum(prices) / len(prices)
    cheapest = rows[0]
    def balance(r):
        price, delivery, rating = r[1], r[2], r[3]
        return (price / avg_price) * 0.5 + (delivery / 14) * 0.25 - (rating / 5) * 0.25
    optimal = min(rows, key=balance)
    remaining = [r for r in rows if r is not cheapest and r is not optimal]
    fastest = min(remaining, key=lambda r: r[2]) if remaining else rows[-1]
    result = []
    for label, r in [("Самый дешёвый", cheapest), ("Оптимальный", optimal), ("Самый быстрый", fastest)]:
        supplier, price, delivery, rating = r
        delta = round((price - min_price) / min_price * 100, 1)
        result.append({"criterion": label, "supplier": supplier, "price": price,
                        "delivery_days": delivery, "rating": rating, "delta_vs_min_pct": delta})
    return result, round(avg_price, 2), min_price

Вкладка «Демо» работает прямо в этом файле — открывать отдельно ничего не нужно.

Код из вкладок «AI-парсер» и «Матчер» — готовые Python-модули для вашего backend (кнопка «Скопировать» → вставить в .py файл).

Что сейчас демо, а что заменить для продакшена

  • AI-извлечение атрибутов — сейчас эвристика (regex+словарь). В проде: реальный вызов Claude/GPT API — в коде парсера уже есть флаг DEMO_MODE и готовая функция под ключ ANTHROPIC_API_KEY. Важно: вызов LLM должен идти с backend, не из браузера — ключ API нельзя хранить в JS на клиенте.
  • База товаров — сейчас 12 демо-позиций в JS-массиве. Для 200 000 SKU нужна БД: PostgreSQL + расширение pg_trgm (триграммный индекс) выдержит нечёткий поиск за миллисекунды, либо Elasticsearch/OpenSearch/Meilisearch.
  • Реальные прайсы поставщиков — пришлите Excel/CSV по каждому поставщику, движок матчера сведёт их в единую таблицу (sku_id, товар, категория, ед.изм, поставщик, цена, срок, рейтинг).
  • OCR рукописных заявок — отдельный шаг перед AI-парсером: фото → текст (Yandex Vision OCR / Google Cloud Vision). Не входит в этот файл, требует отдельной интеграции.

Архитектура полной системы

Фото рукописи → OCR → текст заявки
    → AI-парсер (LLM) → атрибуты (товар, материал, размер, кол-во)
    → Матчинг-движок (индекс + fuzzy + фильтр по атрибутам) → найденный SKU
    → Топ-3 поставщика (дешёвый / оптимальный / быстрый) с аналитикой цены
    → Таблица менеджеру / клиенту
